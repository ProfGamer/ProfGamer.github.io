class TCPServerThread extends Thread {
    private Socket socket;
    private String tId;
    public TCPServerThread(Socket socket) {
        this.socket = socket;
        this.tId = UUID.randomUUID().toString();
        System.out.printf("Client %s is connected with Thread %s, waiting for msg...", socket.getRemoteSocketAddress(), tId);
    }
    @Override
    public void run() {
        try {
            DataInputStream dataInputStream = new DataInputStream(socket.getInputStream());
            while (true) {
                    String data = dataInputStream.readUTF();
                    System.out.println("Thread ID: " + tId);
                    System.out.println(socket.getRemoteSocketAddress());
                    System.out.println(data);
                    System.out.println("-----------------");
                    // Â∞ÜÊé•Êî∂Âà∞ÁöÑÊ∂àÊÅØ, ÂèëÈÄÅÁªôÊâÄÊúâÂú®Á∫øÁöÑÂÆ¢Êà∑Á´Ø
                    sendMsgToAll(data);
            }
        } catch (Exception e) {
            System.out.println(socket.getRemoteSocketAddress() + " is disconnected");
            TCPServer.onlineSockets.remove(socket);
        }
    }

    private void sendMsgToAll(String data) throws IOException {
        for (Socket onlineSocket : TCPServer.onlineSockets) {
            DataOutputStream dos = new DataOutputStream(onlineSocket.getOutputStream());
            dos.writeUTF(data);
            dos.flush();
        }
    }

    public String gettId() {
        return tId;
    }
